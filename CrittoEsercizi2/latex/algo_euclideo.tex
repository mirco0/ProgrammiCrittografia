\begin{align*}
    \text{Verifica per \textbf{$(X^2 +X)$}} \\
    & X^8 + X^4 + X^3 + X + 1  = (X^2 + X)(X^6+X^5+X^4+X^3+1) + 1 \\
    & (X^2 + X) = 1 \cdot (X^2 + X) + 0 \\
    \\
    \text{Verifica per \textbf{$(X^4 +X)$}}\\
            & X^8 + X^4 + X^3 + X + 1  = (X^4 + X)(X^4+X + 1) + (X^3 + X^2 +1) \\
            & (X^4 +X) = (X^3 + X^2 +1)(X+1) + (X^2 +1) \\
            & (X^3 + X^2 +1) = (X^2 +1)(X+1) + X \\
            & (X + 1) = X \cdot 1 + 1 \\
            & X = 1\cdot X + 0 \\
    \\
    \text{Verifica per \textbf{$(X^{16} +X)$}}\\
            & X^8 + X^4 + X^3 + X + 1  = (X^{16} + X)(0) + (X^8 + X^4 + X^3 + X + 1 )\\
            & (X^{16} +X) = (X^8 + X^4 + X^3 + X + 1 )(X^8 + X^4 + X^3 + X) + (X^6 +X^4 +X^3 +X^2)\\
            & (X^8 +X^4 +X^3 +X + 1) = (X^6 +X^4 +X^3 + X^2)(X^2+1) + (X^5 +X^4 +X^2 +X +1)\\
            & (X^6 +X^4 ) = (X^5 +X^4 +X + 1) + (X^2 +1)\\
            & (X^5 +X^4 +X + 1) = (X^2 +1)(X^3+X^2+X+1) + 1\\
            & (X^2 +1) = 1 \cdot (X^2 +1) + 0 \\
    \\
\end{align*}

\noindent Poiché l'ultimo resto non nullo è $1$, si conclude che $\text{MCD}(P, X^{2} - X) = 1$,
$\text{MCD}(P, X^{4} - X) = 1$,
$\text{MCD}(P, X^{16} - X) = 1$.
